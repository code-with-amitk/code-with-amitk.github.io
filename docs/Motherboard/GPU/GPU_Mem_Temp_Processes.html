import pynvml       #1
import time
import logging
from collections import deque

class GPUMonitor:
    def __init__(self, gpu_ids):
        pynvml.nvmlInit()
        self.gpus = [pynvml.nvmlDeviceGetHandleByIndex(i) for i in gpu_ids]
        self.history = {i: deque(maxlen=60) for i in gpu_ids}
        
    def collect_metrics(self):
        metrics = []
        for i, gpu in enumerate(self.gpus):
            util = pynvml.nvmlDeviceGetUtilizationRates(gpu)
            mem = pynvml.nvmlDeviceGetMemoryInfo(gpu)
            temp = pynvml.nvmlDeviceGetTemperature(gpu, pynvml.NVML_TEMPERATURE_GPU)
            
            metric = {
                'gpu_id': i,
                'timestamp': time.time(),
                'gpu_util': util.gpu,
                'mem_util': util.memory,
                'mem_used': mem.used,
                'mem_total': mem.total,
                'temperature': temp,
                'processes': self._get_processes(gpu)
            }
            
            self.history[i].append(metric)
            metrics.append(metric)
            
            # Check for anomalies
            self._check_anomalies(i, metric)
            
        return metrics
    
    def _get_processes(self, gpu):

                })
            except:
                continue
        return processes
    
    def _check_anomalies(self, gpu_id, metric):
        # Check memory leak pattern
        if len(self.history[gpu_id]) == 60:
            mem_trend = [m['mem_used'] for m in self.history[gpu_id]]
            if all(mem_trend[i] < mem_trend[i+1] for i in range(len(mem_trend)-1)):
                logging.warning(f"GPU {gpu_id} shows continuous memory growth - possible memory leak")
        
        # Check temperature threshold
        if metric['temperature'] > 85:
            logging.warning(f"GPU {gpu_id} overheating: {metric['temperature']}C")
            
        # Check utilization anomalies
        if metric['gpu_util'] < 5 and metric['mem_util'] > 80:
            logging.warning(f"GPU {gpu_id} shows memory saturation with low compute utilization")
