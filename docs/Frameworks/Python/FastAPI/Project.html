from app.modules.admin.router import router as admin_router
from app.modules.auth.router import router as auth_router
from app.modules.landing.router import router as landing_router
from app.modules.owner.router import router as owner_router
from app.modules.tags.router import router as tags_router

def create_app() -> FastAPI:
    application.include_router(landing_router)
    application.include_router(tags_router)
    application.include_router(auth_router)
    application.include_router(admin_router)
    application.include_router(owner_router)
    return application

app = create_app()
